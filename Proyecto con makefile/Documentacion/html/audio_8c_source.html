65  * 0 Allow no changes

155  return;

159 
160  if(!(SDL_WasInit(SDL_INIT_AUDIO) & SDL_INIT_AUDIO))
161  {

163  return;
164  }
165 
166  SDL_memset(&(gDevice->want), 0, sizeof(gDevice->want));
167 
168  (gDevice->want).freq = AUDIO_FREQUENCY;
169  (gDevice->want).format = AUDIO_FORMAT;
170  (gDevice->want).channels = AUDIO_CHANNELS;
171  (gDevice->want).samples = AUDIO_SAMPLES;
172  (gDevice->want).callback = audioCallback;
173  (gDevice->want).userdata = calloc(1, sizeof(Audio));
174 
175  global = (Audio *) (gDevice->want).userdata;
176 
177  if(global == NULL)
178  {
179  fprintf(stderr, "[%s: %d]Error: Memory allocation error\n", __FILE__, __LINE__);
180  return;
181  }
182 
183  global->buffer = NULL;
184  global->next = NULL;
185 
186  /* want.userdata = newAudio; */
187  if((gDevice->device = SDL_OpenAudioDevice(NULL, 0, &(gDevice->want), NULL, SDL_AUDIO_ALLOW_CHANGES)) == 0)
188  {
189  fprintf(stderr, "[%s: %d]Warning: failed to open audio device: %s\n", __FILE__, __LINE__, SDL_GetError());
190  }
191  else

196  /* Unpause active audio stream */

287 static inline void playAudio(const char * filename, Audio * audio, uint8_t loop, int volume)
288 {
289  Audio * newAudio;
290 
291  /* Check if audio is enabled */
292  if(!gDevice->audioEnabled)
293  {
294  return;
295  }
296 
297  /* If sound, check if under max number of sounds allowed, else don't play */
298  if(loop == 0)
299  {
300  if(gSoundCount >= AUDIO_MAX_SOUNDS)
301  {
302  return;
303  }
304  else
305  {
306  gSoundCount++;
307  }
308  }
309 
310  /* Load from filename or from Memory */
311  if(filename != NULL)
312  {
313  /* Create new music sound with loop */
314  newAudio = createAudio(filename, loop, volume);
315  }
316  else if(audio != NULL)
317  {
318  newAudio = (Audio *) malloc(sizeof(Audio));
319 
320  if(newAudio == NULL)
321  {
322  fprintf(stderr, "[%s: %d]Fatal Error: Memory allocation error\n", __FILE__, __LINE__);
323  return;
324  }
325 
326  memcpy(newAudio, audio, sizeof(Audio));
327 
328  newAudio->volume = volume;
329  newAudio->loop = loop;
330  newAudio->free = 0;
331  }
332  else
333  {
334  fprintf(stderr, "[%s: %d]Warning: filename and Audio parameters NULL\n", __FILE__, __LINE__);
335  return;
336  }
337 
338  /* Lock callback function */
339  SDL_LockAudioDevice(gDevice->device);
340 
341  if(loop == 1)
342  {
343  addMusic((Audio *) (gDevice->want).userdata, newAudio);
344  }
345  else
346  {
347  addAudio((Audio *) (gDevice->want).userdata, newAudio);
348  }
349 
350  SDL_UnlockAudioDevice(gDevice->device);
351 
352 }

354 static void addMusic(Audio * root, Audio * newAudio)
355 {
356  uint8_t musicFound = 0;
357  Audio * rootNext = root->next;
358 
359  /* Find any existing musics, 0, 1 or 2 and fade them out */
360  while(rootNext != NULL)
361  {
362  /* Phase out any current music */
363  if(rootNext->loop == 1 && rootNext->fade == 0)
364  {
365  if(musicFound)
366  {
367  rootNext->length = 0;
368  rootNext->volume = 0;
369  }
370 
371  rootNext->fade = 1;
372  }
373  /* Set flag to remove any queued up music in favour of new music */
374  else if(rootNext->loop == 1 && rootNext->fade == 1)
375  {
376  musicFound = 1;
377  }
378 
379  rootNext = rootNext->next;
380  }
381 
382  addAudio(root, newAudio);
383 }

385 static inline void audioCallback(void * userdata, uint8_t * stream, int len)
386 {
387  Audio * audio = (Audio *) userdata;
388  Audio * previous = audio;
389  int tempLength;
390  uint8_t music = 0;
391 
392  /* Silence the main buffer */
393  SDL_memset(stream, 0, len);
394 
395  /* First one is place holder */
396  audio = audio->next;
397 
398  while(audio != NULL)
399  {
400  if(audio->length > 0)
401  {
402  if(audio->fade == 1 && audio->loop == 1)
403  {
404  music = 1;
405 
406  if(audio->volume > 0)
407  {
408  if(audio->volume - AUDIO_MUSIC_FADE_VALUE < 0)
409  {
410  audio->volume = 0;
411  }
412  else
413  {
414  audio->volume -= AUDIO_MUSIC_FADE_VALUE;
415  }
416  }
417  else
418  {
419  audio->length = 0;
420  }
421  }
422 
423  if(music && audio->loop == 1 && audio->fade == 0)
424  {
425  tempLength = 0;
426  }
427  else
428  {
429  tempLength = ((uint32_t) len > audio->length) ? audio->length : (uint32_t) len;
430  }
431 
432  SDL_MixAudioFormat(stream, audio->buffer, AUDIO_FORMAT, tempLength, audio->volume);
433 
434  audio->buffer += tempLength;
435  audio->length -= tempLength;
436 
437  previous = audio;
438  audio = audio->next;
439  }
440  else if(audio->loop == 1 && audio->fade == 0)
441  {
442  audio->buffer = audio->bufferTrue;
443  audio->length = audio->lengthTrue;
444  }
445  else
446  {
447  previous->next = audio->next;
448 
449  if(audio->loop == 0)
450  {
451  gSoundCount--;
452  }
453 
454  audio->next = NULL;
455  freeAudio(audio);
456 
457  audio = previous->next;
458  }
459  }
460 }
